local allSkinCards = {}

local function addSkinCard(skinName, sourceFolder)
    local card = Instance.new("Frame", SkinsScroll)
    card.Size = UDim2.new(0, 140, 0, 42)
    card.BackgroundColor3 = Color3.fromRGB(48, 48, 68)
    Instance.new("UICorner", card).CornerRadius = UDim.new(0, 8)
    
    local nameLabel = Instance.new("TextLabel", card)
    nameLabel.Size = UDim2.new(0.6, -5, 1, -4)
    nameLabel.Position = UDim2.new(0, 5, 0, 2)
    nameLabel.BackgroundTransparency = 1
    nameLabel.Text = skinName
    nameLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    nameLabel.TextScaled = true
    nameLabel.Font = Enum.Font.GothamBold
    nameLabel.TextXAlignment = Enum.TextXAlignment.Left
    
    local applyBtn = Instance.new("TextButton", card)
    applyBtn.Size = UDim2.new(0.35, -5, 0.7, -4)
    applyBtn.Position = UDim2.new(0.65, 0, 0.15, 0)
    applyBtn.BackgroundColor3 = Color3.fromRGB(70, 140, 110)
    applyBtn.Text = "Apply"
    applyBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
    applyBtn.TextScaled = true
    applyBtn.Font = Enum.Font.GothamBold
    Instance.new("UICorner", applyBtn).CornerRadius = UDim.new(0, 6)
    
    applyBtn.MouseButton1Click:Connect(function()
        if not selectedWeapon then
            selectedWeaponLabel.Text = "⚠️ Please select a weapon first!"
            selectedWeaponLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
            task.wait(1.5)
            selectedWeaponLabel.Text = "⚡ No weapon selected"
            selectedWeaponLabel.TextColor3 = Color3.fromRGB(255, 180, 80)
            return
        end
        
        local source = sourceFolder:FindFirstChild(skinName)
        if not source then return end
        
        for _, child in pairs(selectedWeapon:GetChildren()) do
            child:Destroy()
        end
        
        for _, child in pairs(source:GetChildren()) do
            child:Clone().Parent = selectedWeapon
        end
        
        local originalColor = applyBtn.BackgroundColor3
        applyBtn.BackgroundColor3 = Color3.fromRGB(100, 200, 100)
        selectedWeaponLabel.Text = "✓ Applied '" .. skinName .. "' to " .. selectedWeapon.Name
        selectedWeaponLabel.TextColor3 = Color3.fromRGB(100, 255, 100)
        task.wait(0.6)
        applyBtn.BackgroundColor3 = originalColor
        task.wait(1)
        selectedWeaponLabel.Text = "✅ Selected: " .. selectedWeapon.Name
        selectedWeaponLabel.TextColor3 = Color3.fromRGB(100, 255, 150)
    end)
    
    return card, nameLabel
end

local skinFolders = {"Bundles","Skin Case","Skin Case 2","Unobtainable","Other","Glorious","Season","Fastive SkinCase"}
for _, folderName in ipairs(skinFolders) do
    local folder = viewModels:FindFirstChild(folderName)
    if folder then
        for _, skin in pairs(folder:GetChildren()) do
            if skin:IsA("Model") or skin:IsA("Folder") then
                local card, nameLabel = addSkinCard(skin.Name, folder)
                table.insert(allSkinCards, {card = card, name = skin.Name:lower(), nameLabel = nameLabel})
            end
        end
    end
end

local function updateCanvasSize()
    task.wait(0.1)
    SkinsScroll.CanvasSize = UDim2.new(0, 0, 0, SkinsGrid.AbsoluteContentSize.Y + 10)
end
updateCanvasSize()

SearchBox:GetPropertyChangedSignal("Text"):Connect(function()
    local searchTerm = SearchBox.Text:lower()
    for _, item in ipairs(allSkinCards) do
        item.card.Visible = (searchTerm == "" or item.name:find(searchTerm))
    end
    updateCanvasSize()
end)
